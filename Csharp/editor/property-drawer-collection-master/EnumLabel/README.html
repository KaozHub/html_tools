EnumLabel¶

Enumの表示名を変更する

###引数

|変数|説明| |---|---| |text|変更する文字列|

###使い方

using UnityEngine;

public class EnumLabelExample : MonoBehaviour
{
    public enum Example
    {
        [EnumLabel("テスト")]
        HIGH,
        [EnumLabel("その２")]
        LOW
    }

    [EnumLabel("例")]
    public Example test = Example.HIGH;
}
